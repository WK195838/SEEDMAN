                        & CPH96 = %RANGE("' || &SYM || '"        +
                           "' || &EYM || '")')                   +
                          KEYFLD(*FILE)                          +

             CHGVAR     VAR(&P6403I) VALUE(&$CPY)

             OVRDBF     FILE(PYEMPF)   SHARE(*YES)

             DLTOVR     FILE(*ALL)

             RTVDTAARA  DTAARA(*LDA (219 1)) RTNVAR(&$PENV)
